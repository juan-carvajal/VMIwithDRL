\subsection{Integer Programming Model}

		
	\textbf{Objective Function}
	\begin{equation}
	\begin{array}{ll@{}ll}
	\min  & \displaystyle\sum\limits_{h \in H}&(I_{h0}CV + F_{h}CF)\\
	\end{array}
	\label{eq:Eq31}
	\end{equation}

	\textbf{Constraints}
	
	\begin{equation}
	\begin{array}{ll@{}ll}
	& \displaystyle I_{h0} = \max(0, I_{h1} + X_{h1} - D_{h})\\
	\end{array}
	\label{eq:Eq32}
	\end{equation}	
		
	\begin{equation}
	\begin{array}{ll@{}ll}
	& \displaystyle F_{h} = \max(0, D_{h} - &\sum\limits_{r \in R}(I_{hr} + X_{hr}))\\
	\end{array}
	\label{eq:Eq33}	
	\end{equation}
	
	\begin{equation}
	\begin{array}{ll@{}ll}
	& \displaystyle F_{h} \leq \frac{1}{\| H \|} &\sum\limits_{h \in H}(F_{h})\\
	\end{array}
	\label{eq:Eq33}	
	\end{equation}
	
	\begin{equation}
	\begin{array}{ll@{}ll}
	& \displaystyle A_{r} = \sum\limits_{h \in H}(X_{hr})\\
	\end{array}
	\label{eq:Eq33}	
	\end{equation}

	\begin{equation}
	\begin{array}{ll@{}ll}
	& \displaystyle \frac{\sum\limits_{r \in R}(I_{hr}+X_{hr})}{D_{h}} = \frac {\sum\limits_{r \in R}(I_{h+1r}+X_{h+1r})}{D_{h+1}} \ \ \forall h<4 \\
	\end{array}
	\label{eq:Eq33}	
	\end{equation}
